int prime[1001] = {1, 1, 0};
/* X计1 借计0 */

void build_board(){
    for(int i = 2; i<=1000; i++){
        if(prime[i] == 1)
            continue;
        /* OX计 L涵考皮wgQ]Θ1 */
        /* 
            i = 4 --> j = 4 8 12 16....
            ぇei = 2寒森 j = 2 4 6 8 10 12 14 16...

            `Nj@}lOi+i --> i*2 --> ゲwOX计 
            CΩj伴Чj = j + i --> j = i*3, i*4, i*5... until 1000

            prime[j] = 1;
            /* NX计害旄m]Θ1 */
        }
    }
}
